Python Repetition:

For:

              
num_str = input("Input an int: \n")
N = int(num_str)
          
for i in range(1,N):
    print(i)

While:

              
num_str = int(input("Input an int: \n"))

num_sum = 0
while num_str != 10:
    num_sum += num_str
    num_str = int(input("Input an int: \n"))
                
print(num_sum)

C++ Repetition:

For:

              
int main(){
    for (int i = 0; i < 5; ++i) cout << i << ", ";
                
    cout << endl;
}

While:

              
int main(){
    int x;
    int sum = 0;
    while (cin >> x){
        sum += x;     
    }
    cout << sum << endl;
}

Python Classes:

              
class Student(object):
    def __init__(self,score = 10):
        self.score = score
                    
    def add_score(self):
        self.score += 10
            
    def decrease_score(self):
        self.score -= 10
                
    def __str__(self):
        current_score = str(self.score)
        return current_score
